1 /** The minplayer namespace. */
  2 minplayer = minplayer || {};
  3 
  4 /**
  5  * @constructor
  6  * @extends minplayer.plugin
  7  * @class Base class used to provide the display and options for any component
  8  * deriving from this class.  Components who derive are expected to provide
  9  * the elements that they define by implementing the getElements method.
 10  *
 11  * @param {string} name The name of this plugin.
 12  * @param {object} context The jQuery context this component resides.
 13  * @param {object} options The options for this component.
 14  */
 15 minplayer.display = function(name, context, options) {
 16 
 17   // See if we allow resize on this display.
 18   this.allowResize = false;
 19 
 20   if (context) {
 21 
 22     // Set the display.
 23     this.display = this.getDisplay(context);
 24   }
 25 
 26   // Derive from plugin
 27   minplayer.plugin.call(this, name, context, options);
 28 };
 29 
 30 /** Derive from minplayer.plugin. */
 31 minplayer.display.prototype = new minplayer.plugin();
 32 
 33 /** Reset the constructor. */
 34 minplayer.display.prototype.constructor = minplayer.display;
 35 
 36 /**
 37  * Returns the display for this component.
 38  *
 39  * @param {object} context The original context.
 40  * @return {object} The jQuery context for this display.
 41  */
 42 minplayer.display.prototype.getDisplay = function(context) {
 43   return jQuery(context);
 44 };
 45 
 46 /**
 47  * @see minplayer.plugin.construct
 48  */
 49 minplayer.display.prototype.construct = function() {
 50 
 51   // Call the plugin constructor.
 52   minplayer.plugin.prototype.construct.call(this);
 53 
 54   // Extend all display elements.
 55   this.options.elements = this.options.elements || {};
 56   jQuery.extend(this.options.elements, this.getElements());
 57   this.elements = this.options.elements;
 58 
 59   // Only do this if they allow resize for this display.
 60   if (this.allowResize) {
 61 
 62     // Set the resize timeout and this pointer.
 63     var resizeTimeout = 0;
 64     var _this = this;
 65 
 66     // Add a handler to trigger a resize event.
 67     jQuery(window).resize(function() {
 68       clearTimeout(resizeTimeout);
 69       resizeTimeout = setTimeout(function() {
 70         _this.onResize();
 71       }, 200);
 72     });
 73   }
 74 };
 75 
 76 /**

 83 /**
 84  * Trigger a media event.
 85  *
 86  * @param {string} type The event type.
 87  * @param {object} data The event data object.
 88  * @return {object} The jQuery prototype.
 89  */
 90 minplayer.display.prototype.trigger = function(type, data) {
 91   return this.display.trigger(type, data);
 92 };
 93 
 94 /**
 95  * Bind to a media event.
 96  *
 97  * @param {string} types The event type.
 98  * @param {object} data The data to bind with the event.
 99  * @param {function} fn The callback function.
100  * @return {object} The jQuery prototype.
101  **/
102 minplayer.display.prototype.bind = function(types, data, fn) {
103 
104   // We will always unbind first for media events.
105   return this.display.unbind(types, fn).bind(types, data, fn);
106 };

108 /**
109  * Returns a scaled rectangle provided a ratio and the container rect.
110  *
111  * @param {number} ratio The width/height ratio of what is being scaled.
112  * @param {object} rect The bounding rectangle for scaling.
113  * @return {object} The Rectangle object of the scaled rectangle.
114  */
115 minplayer.display.prototype.getScaledRect = function(ratio, rect) {
116   var scaledRect = {};
117   scaledRect.x = rect.x ? rect.x : 0;
118   scaledRect.y = rect.y ? rect.y : 0;
119   scaledRect.width = rect.width ? rect.width : 0;
120   scaledRect.height = rect.height ? rect.height : 0;
121   if (ratio) {
122     if ((rect.width / rect.height) > ratio) {
123       scaledRect.height = rect.height;
124       scaledRect.width = Math.floor(rect.height * ratio);
125     }
126     else {
127       scaledRect.height = Math.floor(rect.width / ratio);
128       scaledRect.width = rect.width;
129     }
130     scaledRect.x = Math.floor((rect.width - scaledRect.width) / 2);
131     scaledRect.y = Math.floor((rect.height - scaledRect.height) / 2);
132   }
133   return scaledRect;
134 };
