 56 /**
 57  * @see minplayer.plugin.construct
 58  * @this minplayer.players.base
 59  */
 60 minplayer.players.base.prototype.construct = function() {
 61 
 62   // Call the media display constructor.
 63   minplayer.display.prototype.construct.call(this);
 64 
 65   // Reset the variables to initial state.
 66   this.reset();
 67 
 68   /** The currently loaded media file. */
 69   this.mediaFile = this.options.file;
 70 
 71   // Get the player display object.
 72   if (!this.playerFound()) {
 73 
 74     // Cleanup some events and code.
 75     this.display.unbind();
 76 
 77     // Remove the media element if found
 78     if (this.elements.media) {
 79       this.elements.media.remove();
 80     }
 81 
 82     // Create a new media player element.
 83     this.display.html(this.create());
 84   }
 85 
 86   // Get the player object...
 87   this.player = this.getPlayer();
 88 };

 90 /**
 91  * Clears all the intervals.
 92  */
 93 minplayer.players.base.prototype.clearIntervals = function() {
 94   // Stop the intervals.
 95   if (this.playInterval) {
 96     clearInterval(this.playInterval);
 97   }
 98 
 99   if (this.progressInterval) {
100     clearInterval(this.progressInterval);
101   }
102 };

104 /**
105  * Resets all variables.
106  */
107 minplayer.players.base.prototype.reset = function() {
108 
109   // Reset the ready flag.
110   this.playerReady = false;
111 
112   // The duration of the player.
113   this.duration = new minplayer.async();
114 
115   // The current play time of the player.
116   this.currentTime = new minplayer.async();
117 
118   // The amount of bytes loaded in the player.
119   this.bytesLoaded = new minplayer.async();
120 
121   // The total amount of bytes for the media.
122   this.bytesTotal = new minplayer.async();
123 
124   // The bytes that the download started with.
125   this.bytesStart = new minplayer.async();
126 
127   // The current volume of the player.
128   this.volume = new minplayer.async();
129 
130   // Stop the intervals.
131   this.clearIntervals();
132 
133   // Set the intervals to zero.
134   this.playInterval = 0;
135   this.progressInterval = 0;
136 };

138 /**
139  * Called when the player is ready to recieve events and commands.
140  */
141 minplayer.players.base.prototype.onReady = function() {
142   // Store the this pointer.
143   var _this = this;
144 
145   // Set the ready flag.
146   this.playerReady = true;
147 
148   // Set the volume to the default.
149   this.setVolume(this.options.volume / 100);
150 
151   // Create a progress interval to keep track of the bytes loaded.
152   this.progressInterval = setInterval(function() {
153 
154     // Get the bytes loaded asynchronously.
155     _this.getBytesLoaded(function(bytesLoaded) {
156 
157       // Get the bytes total asynchronously.
158       _this.getBytesTotal(function(bytesTotal) {
159 
160         // Trigger an event about the progress.
161         if (bytesLoaded || bytesTotal) {
162 
163           // Get the bytes start, but don't require it.
164           var bytesStart = 0;
165           _this.getBytesStart(function(val) {
166             bytesStart = val;
167           });
168 
169           // Trigger a progress event.
170           _this.trigger('progress', {
171             loaded: bytesLoaded,
172             total: bytesTotal,
173             start: bytesStart
174           });
175         }
176       });
177 
178     });
179   }, 1000);
180 
181   // We are now ready.
182   this.ready();
183 
184   // Trigger that the load has started.
185   this.trigger('loadstart');
186 };

188 /**
189  * Should be called when the media is playing.
190  */
191 minplayer.players.base.prototype.onPlaying = function() {
192   // Store the this pointer.
193   var _this = this;
194 
195   // Trigger an event that we are playing.
196   this.trigger('playing');
197 
198   // Create a progress interval to keep track of the bytes loaded.
199   this.playInterval = setInterval(function() {
200 
201     // Get the current time asyncrhonously.
202     _this.getCurrentTime(function(currentTime) {
203 
204       // Get the duration asynchronously.
205       _this.getDuration(function(duration) {
206 
207         // Convert these to floats.
208         currentTime = parseFloat(currentTime);
209         duration = parseFloat(duration);
210 
211         // Trigger an event about the progress.
212         if (currentTime || duration) {
213 
214           // Trigger an update event.
215           _this.trigger('timeupdate', {
216             currentTime: currentTime,
217             duration: duration
218           });
219         }
220       });
221     });
222   }, 1000);
223 };
